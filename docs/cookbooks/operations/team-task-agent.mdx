## Overview

Build a multi-user collaborative chat or task management system with Mem0. Each message is attributed to its author, and all messages are stored in a shared project space. Mem0 makes it easy to track contributions, sort and group messages, and collaborate in real time.

## Setup

Install the required packages:

```bash
npm install openai mem0ai
```

## Full Code Example

```typescript
import OpenAI from 'openai';
import { Memory } from 'mem0ai/oss';

// Shared project context
const RUN_ID = 'project-demo';

// Initialize Mem0 and OpenAI
const mem = new Memory();
const client = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

class CollaborativeAgent {
  private runId: string;

  constructor(runId: string) {
    this.runId = runId;
  }

  async addMessage(role: string, name: string, content: string) {
    await mem.add([{ role, name, content }], { runId: this.runId, infer: false });
  }

  async brainstorm(prompt: string): Promise<string> {
    const searchResult = await mem.search(prompt, { runId: this.runId, limit: 5 });
    const context = searchResult.results
      .map((m: any) => `- ${m.memory} (by ${m.actor_id ?? 'Unknown'})`)
      .join('\n');

    const reply = (await client.chat.completions.create({
      model: 'gpt-4.1-nano-2025-04-14',
      messages: [
        { role: 'system', content: 'You are a helpful project assistant.' },
        { role: 'user', content: `Prompt: ${prompt}\nContext:\n${context}` },
      ],
    })).choices[0].message.content?.trim() ?? '';

    await this.addMessage('assistant', 'assistant', reply);
    return reply;
  }

  async getAllMessages() {
    const result = await mem.getAll({ runId: this.runId });
    return result.results;
  }

  async printSortedByTime() {
    const messages = await this.getAllMessages();
    messages.sort((a: any, b: any) => (a.created_at ?? '').localeCompare(b.created_at ?? ''));
    console.log('\n--- Messages (sorted by time) ---');
    for (const m of messages) {
      const ts = m.created_at ? new Date(m.created_at).toISOString().replace('T', ' ').slice(0, 19) : 'N/A';
      console.log(`[${ts}] [${m.actor_id ?? 'Unknown'}] ${m.memory}`);
    }
  }

  async printGroupedByActor() {
    const messages = await this.getAllMessages();
    const grouped: Record<string, any[]> = {};
    for (const m of messages) {
      const actor = m.actor_id ?? 'Unknown';
      (grouped[actor] ??= []).push(m);
    }
    console.log('\n--- Messages (grouped by actor) ---');
    for (const [actor, mems] of Object.entries(grouped)) {
      console.log(`\n=== ${actor} ===`);
      for (const m of mems) {
        const ts = m.created_at ? new Date(m.created_at).toISOString().replace('T', ' ').slice(0, 19) : 'N/A';
        console.log(`[${ts}] ${m.memory}`);
      }
    }
  }
}
```

## Usage

```typescript
// Example usage
const agent = new CollaborativeAgent(RUN_ID);
await agent.addMessage('user', 'alice', "Let's list tasks for the new landing page.");
await agent.addMessage('user', 'bob', "I'll own the hero section copy.");
await agent.addMessage('user', 'carol', "I'll choose product screenshots.");

// Brainstorm with context
const reply = await agent.brainstorm('What are the current open tasks?');
console.log('\nAssistant reply:\n', reply);

// Print all messages sorted by time
await agent.printSortedByTime();

// Print all messages grouped by actor
await agent.printGroupedByActor();
```

## Key Points

- Each message is attributed to a user or agent (actor)
- All messages are stored in a shared project space (`run_id`)
- You can sort messages by time, group by actor, and format timestamps for clarity
- Mem0 makes it easy to build collaborative, attributed chat/task systems

## Conclusion

Mem0 enables fast, transparent collaboration for teams and agents, with full attribution, flexible memory search, and easy message organization.
